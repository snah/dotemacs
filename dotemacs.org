* Misc
** Dice rolling
#+begin_src emacs-lisp
  (defun roll (instr)
    "Roll dice as give in INSTR."
    (interactive "s")
    (message "%s" instr)
    (cond ((equal (string-trim instr) "1d6")
	   (message "[1d6]: %d" (+ 1 (random 6))))
	  ((equal instr "1d10")
	   (message "[1d6]: %d" (+ 1 (random 10))))
	  ((equal instr "1d20")
	   (message "[1d20]: %d" (+ 1 (random 20))))
	  ((equal instr "1d100")
	   (message "[1d100]: %d" (+ 1 (random 100))))))
#+end_src
